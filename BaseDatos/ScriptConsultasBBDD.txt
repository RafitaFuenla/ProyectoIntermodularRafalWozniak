USE BoxRanger;

-- 1. Listar todos los socios
SELECT * 
FROM Socios;

-- 2. Listar todos los entrenadores
SELECT * 
FROM Entrenadores;

-- 3. Mostrar las clases con el nombre del entrenador
SELECT 
    c.idClase,
    c.nombre AS nombre_clase,
    c.hora_inicio,
    c.hora_fin,
    c.max_atletas,
    e.nombre AS nombre_entrenador,
    e.apellidos
FROM Clases c
JOIN Entrenadores e ON c.idEntrenador = e.idEntrenador;

-- 4. Mostrar los pagos realizados por cada socio
SELECT 
    p.idPago,
    s.nombre,
    s.apellidos,
    p.cuota,
    p.fecha_pago,
    p.estado
FROM Pagos p
JOIN Socios s ON p.idSocio = s.idSocio;

-- 5. Mostrar las inscripciones de socios a clases
SELECT 
    s.nombre,
    s.apellidos,
    c.nombre AS nombre_clase,
    i.fecha_inscripcion
FROM Inscripciones i
JOIN Socios s ON i.idSocio = s.idSocio
JOIN Clases c ON i.idClase = c.idClase;

-- 6. Buscar un socio por nombre
SELECT *
FROM Socios
WHERE nombre = 'Ana';

-- 7. Buscar un entrenador por especialidad
SELECT *
FROM Entrenadores
WHERE especialidad = 'CrossFit';

-- 8. Mostrar los pagos pendientes
SELECT 
    p.idPago,
    s.nombre,
    s.apellidos,
    p.cuota,
    p.fecha_pago,
    p.estado
FROM Pagos p
JOIN Socios s ON p.idSocio = s.idSocio
WHERE p.estado = 'pendiente';

-- 9. Mostrar los pagos fallidos
SELECT 
    p.idPago,
    s.nombre,
    s.apellidos,
    p.cuota,
    p.fecha_pago,
    p.estado
FROM Pagos p
JOIN Socios s ON p.idSocio = s.idSocio
WHERE p.estado = 'fallido';

-- 10. Mostrar las clases que imparte un entrenador concreto
SELECT 
    c.nombre,
    c.hora_inicio,
    c.hora_fin,
    e.nombre AS entrenador,
    e.apellidos
FROM Clases c
JOIN Entrenadores e ON c.idEntrenador = e.idEntrenador
WHERE e.nombre = 'Carlos';

-- 11. Contar cuántos socios hay inscritos en cada clase
SELECT 
    c.nombre AS nombre_clase,
    COUNT(i.idSocio) AS total_inscritos
FROM Clases c
LEFT JOIN Inscripciones i ON c.idClase = i.idClase
GROUP BY c.idClase, c.nombre;

-- 12. Mostrar las clases que tienen plazas máximas mayores de 12
SELECT *
FROM Clases
WHERE max_atletas > 12;

-- 13. Mostrar socios dados de alta después de una fecha concreta
SELECT *
FROM Socios
WHERE fecha_alta > '2025-02-01';

-- 14. Mostrar socios que no tienen fecha de baja
SELECT *
FROM Socios
WHERE fecha_baja IS NULL;

-- 15. Mostrar cuántos pagos tiene cada socio
SELECT 
    s.nombre,
    s.apellidos,
    COUNT(p.idPago) AS total_pagos
FROM Socios s
LEFT JOIN Pagos p ON s.idSocio = p.idSocio
GROUP BY s.idSocio, s.nombre, s.apellidos;

-- 16. Mostrar el importe total pagado por cada socio (solo pagos pagados)
SELECT 
    s.nombre,
    s.apellidos,
    SUM(p.cuota) AS total_pagado
FROM Socios s
JOIN Pagos p ON s.idSocio = p.idSocio
WHERE p.estado = 'pagado'
GROUP BY s.idSocio, s.nombre, s.apellidos;

-- 17. Buscar clases cuyo nombre contenga una palabra
SELECT *
FROM Clases
WHERE nombre LIKE '%WOD%';

-- 18. Mostrar los socios inscritos en una clase concreta
SELECT 
    c.nombre AS clase,
    s.nombre,
    s.apellidos,
    i.fecha_inscripcion
FROM Inscripciones i
JOIN Socios s ON i.idSocio = s.idSocio
JOIN Clases c ON i.idClase = c.idClase
WHERE c.nombre = 'WOD Mañana';

-- 19. Mostrar entrenadores que no tienen clases asignadas
SELECT 
    e.nombre,
    e.apellidos,
    e.especialidad
FROM Entrenadores e
LEFT JOIN Clases c ON e.idEntrenador = c.idEntrenador
WHERE c.idClase IS NULL;

-- 20. Mostrar la clase con más inscritos
SELECT 
    c.nombre AS nombre_clase,
    COUNT(i.idSocio) AS total_inscritos
FROM Clases c
LEFT JOIN Inscripciones i ON c.idClase = i.idClase
GROUP BY c.idClase, c.nombre
ORDER BY total_inscritos DESC
LIMIT 1;
